# B2667.단지번호붙이기

```python
from collections import deque
N = int(input())
arr = [list(map(int, input())) for _ in range(N)]
di = [0,1,0,-1]
dj = [1,0,-1,0]
q = deque()
ans_lst = []
for i in range(N):
    for j in range(N):
        if arr[i][j] == 1:
            q.append((i,j))
            arr[i][j] = 0
            cnt = 1
            while q:
                ci, cj = q.popleft()
                for d in range(4):
                    ni = ci + di[d]
                    nj = cj + dj[d]
                    if 0 <= ni < N and 0 <= nj < N and arr[ni][nj] == 1:
                        q.append((ni,nj))
                        arr[ni][nj] = 0
                        cnt += 1
            ans_lst.append(cnt)
ans_lst.sort()
print(len(ans_lst))
print(*ans_lst, sep='\n')
```
