```python
from collections import deque
M, N, H = map(int, input().split())
house = [] #전체 창고
for i in range(H): #높이 만큼 박스를 한층씩 받는다
    box = [list(map(int, input().split())) for _ in range(N)]
    house.append(box)

#첫번째 인덱스가 층을 결정하고 두번째 인덱스가 행을 결정하고 세번째 인덱스가 열을 결정한다
df = [1, -1, 0, 0, 0, 0]
di = [0, 0, 0, 1, 0, -1]
dj = [0, 0, 1, 0, -1, 0]
tomato = deque()
for f in range(H):
    for i in range(N):
        for j in range(M):
            if house[f][i][j] == 1:
                tomato.append((f,i,j))

while tomato:
    f, i, j = tomato.popleft()
    for d in range(6):
        nf = f + df[d]
        ni = i + di[d]
        nj = j + dj[d]
        if 0 <= nf < H and 0 <= ni < N and 0 <= nj < M:
            if house[nf][ni][nj] == 0:
                house[nf][ni][nj] = house[f][i][j] + 1
                tomato.append((nf,ni,nj))

def totoamto():
    ans = 0
    for f in range(H):
        for i in range(N):
            for j in range(M):
                if house[f][i][j] == 0:
                    return -1
                else:
                    if ans < house[f][i][j]:
                        ans = house[f][i][j]
    if ans == 1:
        return 0
    else:
        return ans-1
print(totoamto())
```
