from collections import deque
def fish(si, sj):
    q = deque()
    visited = [[0]*N for _ in range(N)]
    q.append((si,sj,0)) # 0은 거리 초기값
    visited[si][sj] = 1
    fish_where = [] # 물고기 위치 저장할 리스트
    while q:
        ci, cj, dis = q.popleft()
        for d in range(4):
            ni = ci + di[d]
            nj = cj + dj[d]
            if 0 <= ni < N and 0 <= nj < N and visited[ni][nj] != 1:
                if arr[ni][nj] == 0 or arr[ni][nj] == shark_size: # 먹지는 못하고 지나갈 수만 있는 경우
                    q.append((ni, nj, dis+1))
                    visited[ni][nj] = 1
                elif arr[ni][nj] < shark_size: # 먹을 수 있는경우 물고기의 위치를 같이 저장
                    fish_where.append((dis+1, ni, nj))
                    q.append((ni, nj, dis+1))
                    visited[ni][nj] = 1
    if fish_where:
        fish_where.sort()
        return fish_where[0]
    else:
        return -1

N = int(input())
arr = [list(map(int, input().split())) for _ in range(N)]
di = [0, 1, 0, -1]
dj = [1, 0, -1, 0]
shark_size = 2
eat = 0
# 아기상어 위치(si, sj), 찾으면 0으로 초기화
for i in range(N):
    for j in range(N):
        if arr[i][j] == 9:
            arr[i][j] = 0
            si, sj = i, j
ans = 0
while 1:
    abc = fish(si, sj)
    if abc != -1:
        si, sj = abc[1], abc[2]
        arr[si][sj] = 0
        eat += 1
        ans += abc[0]
        if eat == shark_size:
            shark_size += 1
            eat = 0
    else:
        break
print(ans)
